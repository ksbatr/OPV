#define _CRT_SECURE_NO_WARNINGS  

#include <iostream>
#include <queue>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <chrono>
#include <vector>
#include <sstream>
#include <iomanip>


std::queue<int> tasks;                      
std::mutex queue_mutex;                   
std::condition_variable cv;                 
bool done = false;                          
 
std::recursive_mutex log_mutex;

thread_local int local_counter = 0;

void log_message(const std::string& msg) {
    std::lock_guard<std::recursive_mutex> lock(log_mutex);
    std::cout << msg << std::endl;
}

void log_with_timestamp(const std::string& msg) {
    std::lock_guard<std::recursive_mutex> lock(log_mutex);

    auto now = std::chrono::system_clock::now();
    auto now_time_t = std::chrono::system_clock::to_time_t(now);

    std::tm time_info;
    localtime_s(&time_info, &now_time_t);

    std::stringstream ss;
    ss << "[" << std::put_time(&time_info, "%H:%M:%S") << "] " << msg;

    log_message(ss.str());  
}

void producer(int task_count) {
    for (int i = 1; i <= task_count; ++i) {
        {
            std::lock_guard<std::mutex> lock(queue_mutex);  
            tasks.push(i);  
            log_with_timestamp("Производитель добавил задачу: " + std::to_string(i));
        }  

        cv.notify_one();

        std::this_thread::sleep_for(std::chrono::milliseconds(100));
    }

    {
        std::lock_guard<std::mutex> lock(queue_mutex);
        done = true;  
        log_with_timestamp("Производитель завершил работу");
    }

    cv.notify_all();
}

void consumer(int id) {
    local_counter = 0;

    while (true) {
        int task = -1;

        {
            std::unique_lock<std::mutex> lock(queue_mutex);  

            cv.wait(lock, [] {
                return !tasks.empty() || done;  
                });

            if (!tasks.empty()) {
                task = tasks.front();  
                tasks.pop();           

                lock.unlock();  

            }
            else if (done) {
                break;  
            }
        }

        if (task != -1) {
            std::stringstream ss;
            ss << std::this_thread::get_id();  
            std::string thread_id = ss.str();

            log_with_timestamp("Потребитель " + std::to_string(id) +
                " (ID потока: " + thread_id +
                ") обрабатывает задачу: " + std::to_string(task));

            int result = task * task;

            local_counter++;

            std::this_thread::sleep_for(std::chrono::milliseconds(150));

            log_with_timestamp("Потребитель " + std::to_string(id) +
                " завершил задачу " + std::to_string(task) +
                " -> результат: " + std::to_string(result));
        }
    }

    log_with_timestamp("Потребитель " + std::to_string(id) +
        " обработал " + std::to_string(local_counter) + " задач");
}

void demonstrate_recursive_mutex() {

    log_with_timestamp("Первычный захват recursive_mutex");
    log_with_timestamp("Повторный захват recursive_mutex");
}


int main() {
    setlocale(LC_ALL, "Russian");

    const int task_count = 20;      // количество задач
    const int consumer_count = 3;   // количество потребителей

    // создаем поток-производитель
    std::thread producer_thread(producer, task_count);

    // создаем несколько потоков-потребителей
    std::vector<std::thread> consumers;
    for (int i = 0; i < consumer_count; ++i) {
        consumers.emplace_back(consumer, i + 1);
    }

    // ожидаем завершения всех потоков
    producer_thread.join();
    for (auto& t : consumers) {
        t.join();
    }

    demonstrate_recursive_mutex();

    return 0;
}
