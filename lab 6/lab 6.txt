#include <omp.h>
#include <opencv2/opencv.hpp>
#include <iostream>
#include <chrono>
#include <vector>
#include <complex>
#include <iomanip>

const double X_MIN = -2.5;
const double X_MAX = 1.0;
const double Y_MIN = -1.5;
const double Y_MAX = 1.5;

const int WIDTH = 1000;
const int HEIGHT = 1000;

const int MAX_ITER = 256;

int mandelbrotIterations(double c_real, double c_imag) {
    std::complex<double> c(c_real, c_imag);
    std::complex<double> z(0, 0);
    int iter = 0;

    while (std::abs(z) < 2.0 && iter < MAX_ITER) {
        z = z * z + c;
        iter++;
    }

    return iter;
}

cv::Vec3b getColor(int iterations) {
    if (iterations == MAX_ITER) {
        return cv::Vec3b(1, 0, 0);
    }

    double t = static_cast<double>(iterations) / MAX_ITER;

    int r = static_cast<int>(9 * (1 - t) * t * t * t * 255);
    int g = static_cast<int>(15 * (1 - t) * (1 - t) * t * t * 255);
    int b = static_cast<int>(8.5 * (1 - t) * (1 - t) * (1 - t) * t * 255);

    return cv::Vec3b(b, g, r);
}

void computeMandelbrotSequential(cv::Mat& image) {
    double step_x = (X_MAX - X_MIN) / WIDTH;
    double step_y = (Y_MAX - Y_MIN) / HEIGHT;

    for (int y = 0; y < HEIGHT; y++) {
        double c_imag = Y_MAX - y * step_y;

        for (int x = 0; x < WIDTH; x++) {
            double c_real = X_MIN + x * step_x;

            int iterations = mandelbrotIterations(c_real, c_imag);
            image.at<cv::Vec3b>(y, x) = getColor(iterations);
        }
    }
}

void computeMandelbrotParallel(cv::Mat& image) {
    double step_x = (X_MAX - X_MIN) / WIDTH;
    double step_y = (Y_MAX - Y_MIN) / HEIGHT;

#pragma omp parallel for schedule(dynamic)
    for (int y = 0; y < HEIGHT; y++) {
        double c_imag = Y_MAX - y * step_y;

        for (int x = 0; x < WIDTH; x++) {
            double c_real = X_MIN + x * step_x;

            int iterations = mandelbrotIterations(c_real, c_imag);
            image.at<cv::Vec3b>(y, x) = getColor(iterations);
        }
    }
}

int main() {
    setlocale(LC_ALL, "Russian");


    std::cout << "Размер изображения: " << WIDTH << "x" << HEIGHT << std::endl;
    std::cout << "Максимум итераций: " << MAX_ITER << std::endl;
    std::cout << "Область: x=[" << X_MIN << ", " << X_MAX << "], y=[" << Y_MIN << ", " << Y_MAX << "]" << std::endl;
    std::cout << std::endl;

    std::cout << "Выполнение последовательной версии" << std::endl;
    cv::Mat image_seq(HEIGHT, WIDTH, CV_8UC3);

    auto start_seq = std::chrono::high_resolution_clock::now();
    computeMandelbrotSequential(image_seq);
    auto end_seq = std::chrono::high_resolution_clock::now();

    double time_seq = std::chrono::duration<double>(end_seq - start_seq).count();
    std::cout << "Время выполнения: " << time_seq << " секунд" << std::endl;
    std::cout << std::endl;



    std::cout << "Выполнение параллельной версии" << std::endl;
    cv::Mat image_par(HEIGHT, WIDTH, CV_8UC3);

    auto start_par = std::chrono::high_resolution_clock::now();
    computeMandelbrotParallel(image_par);
    auto end_par = std::chrono::high_resolution_clock::now();

    double time_par = std::chrono::duration<double>(end_par - start_par).count();
    double speedup = time_seq / time_par;

    std::cout << "Время выполнения: " << time_par << " секунд" << std::endl;

    std::cout << std::endl;


    cv::namedWindow("Множество Мандельброта", cv::WINDOW_NORMAL);
    cv::resizeWindow("Множество Мандельброта", 800, 800);
    cv::imshow("Множество Мандельброта", image_seq);
    cv::waitKey(0);

    std::cout << std::endl;

    return 0;
}
