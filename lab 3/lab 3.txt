#include <iostream>
#include <thread>
#include <future>
#include <vector>
#include <mutex>

std::mutex cout_mutex;

// Вычислительная задача: факториал
unsigned long long factorial(int n) {
    unsigned long long result = 1;
    for (int i = 2; i <= n; ++i) {
        result *= i;
    }
    return result;
}

int main() {
    setlocale(LC_ALL, "Russian");
    const int N = 5;
    std::vector<int> inputs = { 5, 7, 10, 4, 6 };

    std::vector<std::packaged_task<unsigned long long(int)>> tasks;
    std::vector<std::future<unsigned long long>> futures;

    std::promise<void> monitorPromise;
    std::future<void> monitorFuture = monitorPromise.get_future();

    std::vector<std::thread> workerThreads;

    for (int i = 0; i < N; ++i) {
        std::packaged_task<unsigned long long(int)> task(factorial);
        futures.push_back(task.get_future());  
        tasks.push_back(std::move(task));      
    }

    for (int i = 0; i < N; ++i) {
        workerThreads.emplace_back(std::move(tasks[i]), inputs[i]);
    }

    std::thread monitorThread([&]() {
        {
            std::lock_guard<std::mutex> lock(cout_mutex);
            std::cout << "[Монитор] Ожидание завершения всех задач...\n";
        }

        for (auto& t : workerThreads) {
            if (t.joinable()) {
                t.join();
            }
        }

        monitorPromise.set_value();

        std::lock_guard<std::mutex> lock(cout_mutex);
        std::cout << "[Монитор] Все задачи завершены, сигнал отправлен.\n";
        });

    {
        std::lock_guard<std::mutex> lock(cout_mutex);
        std::cout << "[Основной] Ожидание результатов...\n";
    }

    for (int i = 0; i < N; ++i) {
        unsigned long long result = futures[i].get();

        std::lock_guard<std::mutex> lock(cout_mutex);
        std::cout << "Результат задачи " << i + 1 << " (" << inputs[i] << "!): " << result << std::endl;
    }

    {
        std::lock_guard<std::mutex> lock(cout_mutex);
        std::cout << "[Основной] Ожидание сигнала от монитора...\n";
    }

    monitorFuture.wait(); 

    {
        std::lock_guard<std::mutex> lock(cout_mutex);
        std::cout << "[Основной] Получен сигнал от монитора. Завершение.\n";
    }

    if (monitorThread.joinable()) {
        monitorThread.join();
    }

    return 0;
}
